  关键改动：

     ```typescript
     const handleOptionClick = (optionId: string) => {
       if (type === 'single') {
         // 单选题：点击后立即提交
         setSelectedOptions([optionId])
         setShowResult(true)  // 立即显示结果
       } else {
         // 多选题：切换选择状态
         // ... 允许多选
       }
     }
     ```

  2. 按钮显示逻辑
     ```typescript
     {type === 'multiple' && (
       <button onClick={handleSubmitMultiple}>
         提交答案（多选题）
       </button>
     )}
     {type === 'single' && (
       <div>💡 提示：点击选项后将立即显示结果</div>
     )}
     ```

